Configurable network host

Often we have different base urls for production or development, like api.example.com and http://localhost:8080 respectively. If these never change it's pretty easy to configure them in a BaseResource that all resources extend from.

However, it might be nice to be able to quickly switch between testing a local API endpoint, production and staging at the flip of a switch. In that case we'd need some controls in the application itself that dynamically select how urls resolve.

Example Network Flux Control

Let's start by building our simple flux store for our network endpoint:

NetworkContext.ts

import { createContext } from 'react';
import { ActionTypes } from '~/types';

const defaultUrl = (process.env.NODE_ENV === "production") ? 'https://api.example.com' : 'http://localhost:8080';

export const StateContext = createContext(defaultUrl);

export const DispatchContext = createContext((value: ActionTypes) => {
  if (process.env.NODE_ENV !== 'production') {
    console.error(
      'It appears you are trying to use Rest Hooks without a provider.\nFollow instructions: https://resthooks.io/docs/getting-started/installation#add-provider-at-top-level-component',
    );
    if (process.env.NODE_ENV === 'test') {
      console.error(
        'If you are trying to test: https://resthooks.io/docs/guides/unit-testing-hooks',
      );
    }
  }
  return Promise.resolve();
});

NetworkProvider.tsx

import { usePersistedState } from "@urkellabs/ucl";

export default function useNetwork() {
  let [network, setNetwork] = usePersistedState("networkURL", () => {
    if (process.env.NODE_ENV !== "production") {
      return "http://localhost:8080";
    } else {
      return "https://api.hnscan.com";
    }
  });

  return [network, setNetwork];
}

NetworkSwitcher.tsx

import React from "react";
import { Select } from "@urkellabs/ucl";

// Hooks
import useNetwork from "hooks/useNetwork";

export default function NetworkSwitcher(props) {
  let [network, setNetwork] = useNetwork();

  let options = [
    { value: "http://localhost:8080", label: "Local Testnet" },
    { value: "https://api.hnscan.com", label: "HNScan Testnet" },
    { value: "http://", label: "Experimental Testnet" }
  ];

  return (
    <Select
      options={options}
      defaultValue={
        options.find(element => element.value === network) || options[0]
      }
      onChange={kv => {
        setNetwork(kv.value);
      }}
    />
  );
}
